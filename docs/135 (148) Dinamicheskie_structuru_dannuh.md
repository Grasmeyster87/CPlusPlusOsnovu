class Stack {
    int* data;
    int top;
    int capacity;

public:
    Stack(int cap = 10) : top(-1), capacity(cap) {
        data = new int[capacity];
    }

    void push(int x) {
        if (top + 1 == capacity) return; // –±–µ–∑ —Ä–æ–∑—à–∏—Ä–µ–Ω–Ω—è
        data[++top] = x;
    }

    void pop() {
        if (top >= 0) top--;
    }

    int peek() {
        return data[top];
    }

    bool empty() {
        return top == -1;
    }

    ~Stack() {
        delete[] data;
    }
};
```

    int* arr;
    int head, tail, size, capacity;

public:
    Queue(int cap = 10) : head(0), tail(0), size(0), capacity(cap) {
        arr = new int[cap];
    }

    void enqueue(int x) {
        if (size == capacity) return; // –±–µ–∑ —Ä–æ–∑—à–∏—Ä–µ–Ω–Ω—è
        arr[tail] = x;
        tail = (tail + 1) % capacity;
        size++;
    }

    void dequeue() {
        if (size == 0) return;
        head = (head + 1) % capacity;
        size--;
    }

    int front() {
        return arr[head];
    }

    bool empty() {
        return size == 0;
    }

    ~Queue() {
        delete[] arr;
    }
};
```
